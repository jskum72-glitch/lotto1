        // Firebase 초기화
        const app = initializeApp(firebaseConfig);

        // 로또 번호 생성 및 저장 함수
        async function generateLotto() {
            const luckyNumInput = document.getElementById('luckyNumber');
            const luckyNum = parseInt(luckyNumInput.value);
            const resultContainer = document.getElementById('resultContainer');
            const timeDisplay = document.getElementById('timeDisplay');
            const btn = document.getElementById('startBtn');

            // 유효성 검사
            if (isNaN(luckyNum) || luckyNum < 1 || luckyNum > 45) {
                alert("1부터 45 사이의 행운의 숫자를 입력해주세요.");
                return;
            }

            // 버튼 로딩 상태
            btn.disabled = true;
            btn.innerText = "번호 생성 중...";
            resultContainer.innerHTML = "";

            const allGames = [];
            const now = new Date();
            const timeStr = now.toLocaleString();

            // 5행의 로또 번호 생성
            for (let i = 0; i < 5; i++) {
                let numbers = new Set();
                
                // 요구사항: 행운의 숫자가 4번 이상 등장해야 함
                // 5행 중 첫 4행에는 행운의 숫자를 무조건 포함
                if (i < 4) {
                    numbers.add(luckyNum);
                }

                while (numbers.size < 6) {
                    let r = Math.floor(Math.random() * 45) + 1;
                    numbers.add(r);
                }

                const sortedNumbers = Array.from(numbers).sort((a, b) => a - b);
                allGames.push(sortedNumbers);
            }

            // 화면에 실시간 표시 (애니메이션 효과)
            timeDisplay.innerText = `생성 시간: ${timeStr}`;
            
            for (let i = 0; i < allGames.length; i++) {
                await new Promise(resolve => setTimeout(resolve, 200)); // 시각적 효과를 위한 지연
                const rowDiv = document.createElement('div');
                rowDiv.className = "flex justify-center gap-2 mb-4 animate-bounce-short";
                
                allGames[i].forEach(num => {
                    const ball = document.createElement('span');
                    ball.className = `w-10 h-10 md:w-12 md:h-12 flex items-center justify-center rounded-full text-white font-bold shadow-lg ${getBallColor(num)}`;
                    ball.innerText = num;
                    rowDiv.appendChild(ball);
                });
                resultContainer.appendChild(rowDiv);
            }

            // Firestore에 데이터 저장
            try {
                await addDoc(collection(db, "lotto_history"), {
                    lucky_number: luckyNum,
                    generated_numbers: allGames,
                    timestamp: serverTimestamp(),
                    display_time: timeStr
                });
                console.log("데이터가 성공적으로 저장되었습니다.");
            } catch (e) {
                console.error("데이터 저장 오류: ", e);
            }

            btn.disabled = false;
            btn.innerText = "로또 번호 생성 시작";
        }

        // 로또 공 색상 결정 함수
        function getBallColor(num) {
            if (num <= 10) return "bg-yellow-400";
            if (num <= 20) return "bg-blue-400";
            if (num <= 30) return "bg-red-400";
            if (num <= 40) return "bg-gray-500";
            return "bg-green-500";
        }

        // 전역 함수로 등록
        window.generateLotto = generateLotto;
